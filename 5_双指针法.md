```cpp
class Solution {
public:
    int removeElement(vector<int>& nums, int val) {
        // 快慢指针法
        // int left=0, right=0;
        // for(;right<nums.size();right++){
        //     if(val != nums[right]){
        //         nums[left] = nums[right];
        //         left++;
        //     }
        // }
        // 两端指针法
        int left=0, right=nums.size()-1;
        while(left <= right){
            if(val == nums[left]){
                while(nums[right] == val) right--;
                nums[left] = nums[right];
                right--;
            }
            left++;
        }
        return left;
    }
};
```
